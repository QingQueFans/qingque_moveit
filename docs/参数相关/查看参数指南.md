# 通常在这个位置（根据ROS2版本不同）：
# 方法1: 使用ros2命令
ros2 pkg prefix moveit_resources_panda_moveit_config

# 得到类似：/opt/ros/humble/share/moveit_resources_panda_moveit_config
# 然后：
cd /opt/ros/humble/share/moveit_resources_panda_moveit_config
ls -la

# 关键目录：
ls -la config/      # MoveIt配置
ls -la launch/      # 启动文件
# 查看夹爪MoveIt控制器配置
cat gripper_moveit_controllers.yaml

# 查看ROS2控制器配置
cat ros2_controllers.yaml

# 查看手部配置
cat hand.xacro

# 关节限制
cat joint_limits.yaml

# 查找Panda的URDF/XACRO文件
find /opt/ros/humble -name "panda.urdf" -o -name "panda.xacro" 2>/dev/null

# 或者在moveit_resources包中查找
ros2 pkg prefix moveit_resources_panda_description
cd /opt/ros/humble/share/moveit_resources_panda_description
ls -la

# 查看夹爪相关的部分
cat /opt/ros/humble/share/moveit_resources_panda_description/urdf/panda.urdf | \
  grep -n -A5 -B2 "panda_finger\|finger_joint\|leftfinger\|rightfinger"


  # 查看机械臂关节（panda_joint1 到 panda_joint7）
cat /opt/ros/humble/share/moveit_resources_panda_description/urdf/panda.urdf | \
  grep -A10 '<joint name="panda_joint' | \
  grep -E 'name="panda_joint|effort=|lower=|upper=|velocity='
